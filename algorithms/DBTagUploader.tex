\begin{algorithm}

\caption{Upload tags to DynamoDB}\label{euclid}

\begin{algorithmic}[1]
\Procedure{UploadTaggedArticles}{}
\State $\textit{parsedArticles} \gets \textit{map}(\textit{addTagIdToArticle}, \textit{articles})$
\State $\textit{shuffledTags} \gets \textit{Empty List}()$
\BState \emph{topmain}:
\If {\textit{filter} (\textit{empty?}, \textit{parsedArticles})} \Return false
\EndIf
\BState \emph{mainloop}:
\BState \emph{topsub}:
\If {$i >= \textit{parsedArticles}.\textit{length}$} \Return false
\EndIf
\BState \emph{subloop}:
\If {$\textit{string}(i) = \textit{path}(j)$}
\State $j \gets j-1$.
\State $i \gets i-1$.
\State \textbf{goto} \emph{subloop}.
\State \textbf{close};
\EndIf
\State $i \gets i+\max(\textit{delta}_1(\textit{string}(i)),\textit{delta}_2(j))$.
\State \textbf{goto} \emph{topsub}.
\EndProcedure

\end{algorithmic}

\end{algorithm}
